63 color_type_t getBitmapSize(FIBITMAP* pImg) {
64  // Get the data type of the image. Convert to an internal format
65  const int storageType = FreeImage_GetImageType(pImg);
66  color_type_t dataType = COLOR_TYPE_DEFAULT;
67 
68  if (storageType == FIT_UNKNOWN

70  || storageType == FIT_COMPLEX
71  ) {
72  return COLOR_TYPE_INVALID;
73  }
74 
75  switch(storageType) {
76  // n-bit char
77  case FIT_BITMAP:
78  LS_LOG_MSG("\tImage pixel type: BYTE");
79  dataType = COLOR_TYPE_UNSIGNED_BYTE;
80  break;
81

250 bool imageResource::loadFile(const std::string& filename) {
251  LS_LOG_MSG("Attempting to load the image ", filename);
252  unload();
253 
254  if (!filename.size()) {
255  LS_LOG_ERR("\tFailed to load an image as no filename was provided.\n");
256  return false;
257  }
258 
259  // Set FreeImage's error function
260  FreeImage_SetOutputMessage(&printImageLoadError);
261 
262  // Determine the file type that should be loaded
263  FREE_IMAGE_FORMAT fileFormat = deduceImageFormat(filename.c_str());
264 
265  if (fileFormat == FIF_UNKNOWN) {
266  LS_LOG_ERR("\tUnable to determine the file type for ", filename, ".\n");
267  return false;
268  }
269 
270  if (FreeImage_FIFSupportsReading(fileFormat) == false) {
271  LS_LOG_ERR(
272  "\tSupport for the type of file used by ", filename,
273  " is not currently implemented.\n"
274  );
275  return false;
276  }
277 
278  // Preliminary setup passed. Attempt to load the file data
279 
280  // Use some predefined image flags
281  const int fileFlags = getImageFlags(fileFormat);
282  FIBITMAP* fileData = FreeImage_Load(fileFormat, filename.c_str(), fileFlags);
283 
284  if (!fileData) {
285  LS_LOG_ERR(
286  "\tUnable to load the image ", filename,
287  " due to an internal library error.\n"
288  );
289  return false;
290  }
291 
292  const color_type_t dataType = getBitmapSize(fileData);
293  if (dataType == COLOR_TYPE_INVALID) {
294  LS_LOG_ERR('\t', filename, " contains an unsupported pixel format.\n");
295  FreeImage_Unload(fileData);
296  return false;
297  }
298 
299  this->pData = reinterpret_cast<char*>(fileData);
300  this->imgSize[0] = (int)FreeImage_GetWidth(fileData);
301  this->imgSize[1] = (int)FreeImage_GetHeight(fileData);
302  this->bitsPerPixel = (unsigned)FreeImage_GetBPP(fileData);
303  this->pixelType = dataType;
304  this->dataSize = this->imgSize[0] * this->imgSize[1];
305 
306  getPixelFormat(fileData, this->bitsPerPixel, intFormat, extFormat);
307 
308  LS_LOG_MSG("\tSuccessfully loaded ", filename, ".\n");
309 
310  return true;
311 }
