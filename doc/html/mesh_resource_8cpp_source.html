334 bool meshResource::loadCylinder(unsigned numSides) {
335  // make sure there are enough points for a minimal cylinder
336  if (numSides < 2) {
337  numSides = 2;
338  }
339 
340  LS_LOG_MSG("Attempting to load a ", numSides, "-sided cylinder.");
341 
342  if (!initVertices(numSides*12)) {
343  LS_LOG_ERR("\tAn error occurred while initializing a ", numSides, "-sided cylinder.\n");
344  return false;
345  }
346 
347  int iNumSides = (int)numSides;
348  vertex* pCapVert = pVertices;
349  vertex* pSideVert = pVertices+(numSides*6);
350 
351  // Make the cylinder caps using the "makePolygon()" algorithm
352  for (int i = -iNumSides; i < iNumSides; ++i) {
353  const float topBot = (i < 0) ? 1.f : -1.f;
354 
355  const float theta1 = LS_TWO_PI * ((float)i / (float)iNumSides) * topBot;
356  const float bc1 = std::cos(theta1);
357  const float bs1 = std::sin(theta1);
358 
359  const float theta2 = LS_TWO_PI * ((float)(i-1) / (float)iNumSides) * topBot;
360  const float bc2 = std::cos(theta2);
361  const float bs2 = std::sin(theta2);
362 
363  // center
364  pCapVert->pos = math::vec3{0.f, topBot, 0.f};
365  pCapVert->uv = math::vec2{0.5f, 0.5f};
366  pCapVert->norm = math::vec3{0.f, topBot, 0.f};
367  meshBounds.compareAndUpdate(pCapVert->pos);
368  ++pCapVert;
369 
370  // cap, first triangle leg
371  pCapVert->pos = math::vec3{bc1, topBot, bs1};
372  pCapVert->uv = math::vec2{(bs1*0.5f)+0.5f, (bc1*0.5f)+0.5f};
373  pCapVert->norm = math::vec3{0.f, topBot, 0.f};
374  meshBounds.compareAndUpdate(pCapVert->pos);
375  ++pCapVert;
376 
377  // cap, second triangle leg
378  pCapVert->pos = math::vec3{bc2, topBot, bs2};
379  pCapVert->uv = math::vec2{(bs2*0.5f)+0.5f, (bc2*0.5f)+0.5f};
380  pCapVert->norm = math::vec3{0.f, topBot, 0.f};
381  meshBounds.compareAndUpdate(pCapVert->pos);

386  pSideVert->uv = math::vec2{(bs1*0.5f)+0.5f, (bc1*0.5f)+0.5f};

478 /*-------------------------------------
479  Load a Sphere
480 
481  I found this method on the website by Kevin Harris:
482  http://www.codesampler.com/oglsrc/oglsrc_8.htm#ogl_textured_sphere
483 
484  This loading method was originally found here:
485  http://astronomy.swin.edu.au/~pbourke/opengl/sphere/
486 -------------------------------------*/
487 bool meshResource::loadSphere(unsigned res) {
488  // Only load an even number of vertices
489  if (res % 2 == 1) {
490  ++res;
491  }
492 
493  // calculate the exact number of vertices to load
494  unsigned totalVertCount = res*(res+1); // more trial and error
495 
496  LS_LOG_MSG("Attempting to load a ", totalVertCount, "-point sphere (", res, "x).");
497 
498  if (!initVertices(totalVertCount)) {
499  LS_LOG_ERR("\tAn error occurred while initializing a ", totalVertCount, "-point sphere.\n");
500  return false;
501  }
502 
503  const int iNumSides = (int)res;
504  const float fNumSides = (float)res;
505  vertex* pVert = pVertices;
506 
507  for(int i = 0; i < iNumSides/2; ++i) {
508  float theta1 = i * LS_TWO_PI / fNumSides - LS_PI_OVER_2;
509  float theta2 = (i + 1) * LS_TWO_PI / fNumSides - LS_PI_OVER_2;
510 
511  for(int j = 0; j <= iNumSides; ++j) {
512  const float theta3 = j * LS_TWO_PI / fNumSides;
513 
514  {
515  const float ex = LS_COS(theta1) * LS_SIN(theta3);
516  const float ey = LS_SIN(theta1);
517  const float ez = LS_COS(theta1) * LS_COS(theta3);
518  pVert->pos = math::vec3{ex, ey, -ez};
519  pVert->uv = math::vec2{-j/fNumSides, 2.f*i/fNumSides};
520  pVert->norm = pVert->pos;
521 
522  meshBounds.compareAndUpdate(pVert->pos);
523  ++pVert;
524  }
525  {
526  const float ex = LS_COS(theta2) * LS_SIN(theta3);
527  const float ey = LS_SIN(theta2);
528  const float ez = LS_COS(theta2) * LS_COS(theta3);
529  pVert->pos = math::vec3{ex, ey, -ez};
530  pVert->uv = math::vec2{-j/fNumSides , 2.f*(i+1)/fNumSides};
531  pVert->norm = pVert->pos;
532 
533  meshBounds.compareAndUpdate(pVert->pos);
534  ++pVert;
535  }
536  }
537  }
538 
539  LS_LOG_MSG("\tSuccessfully loaded a ", totalVertCount, "-point sphere.\n");
540  resultDrawMode = draw_mode_t::DRAW_MODE_TRI_STRIP;
541  return true;
542 }
